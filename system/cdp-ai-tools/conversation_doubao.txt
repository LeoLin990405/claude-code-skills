import os
import argparse

def factorial(n):
    """阶乘计算（核心功能1）"""
    if not isinstance(n, int):
        raise TypeError("阶乘参数必须是整数")
    if n < 0:
        raise ValueError("阶乘参数不能为负数")
    if n == 0 or n == 1:
        return 1
    return n * factorial(n - 1)

def traverse_folder(folder_path):
    """文件夹遍历（核心功能2）"""
    if not os.path.exists(folder_path):
        raise FileNotFoundError(f"文件夹不存在：{folder_path}")
    file_list = []
    for item in os.listdir(folder_path):
        item_path = os.path.join(folder_path, item)
        if os.path.isfile(item_path):
            file_list.append(f"文件：{item_path}")
        elif os.path.isdir(item_path):
            file_list.append(f"文件夹：{item_path}")
            file_list.extend(traverse_folder(item_path))
    return file_list

def main():

if __name__ == "__main__":

import os
import argparse

def factorial(n):
    """阶乘计算（核心功能1）"""
    if not isinstance(n, int):
        raise TypeError("阶乘参数必须是整数")
    if n < 0:
        raise ValueError("阶乘参数不能为负数")
    if n == 0 or n == 1:
        return 1
    return n * factorial(n - 1)

def traverse_folder(folder_path):
    """文件夹遍历（核心功能2）"""
    if not os.path.exists(folder_path):
        raise FileNotFoundError(f"文件夹不存在：{folder_path}")
    file_list = []
    for item in os.listdir(folder_path):
        item_path = os.path.join(folder_path, item)
        if os.path.isfile(item_path):
            file_list.append(f"文件：{item_path}")
        elif os.path.isdir(item_path):
            file_list.append(f"文件夹：{item_path}")
            file_list.extend(traverse_folder(item_path))
    return file_list

def main():
    # 命令行参数解析（易用性优化）
    parser = argparse.ArgumentParser(description="统一工具最终版：阶乘计算+文件夹遍历")
    subparsers = parser.add_subparsers(dest="command", help="子命令：factorial（阶乘）/traverse（遍历文件夹）")
    
    # 阶乘子命令
    fact_parser = subparsers.add_parser("factorial", help="计算阶乘")
    fact_parser.add_argument("n", type=int, help="要计算的数（非负整数）")
    
    # 文件夹遍历子命令
    traverse_parser = subparsers.add_parser("traverse", help="遍历文件夹")
    traverse_parser.add_argument("path", help="文件夹路径")
    
    args = parser.parse_args()
    
    # 执行对应功能
    try:
        if args.command == "factorial":
            result = factorial(args.n)
            print(f"{args.n}的阶乘是：{result}")
        elif args.command == "traverse":
            result = traverse_folder(args.path)
            print("遍历结果：")
            for line in result:
                print(line)
        else:
            parser.print_help()
    except Exception as e:
        print(f"执行错误：{e}")  # 友好异常处理

if __name__ == "__main__":
    main()
